    防火墙配置¶

    Markdown语法介绍

    • 查看防火墙状态
    systemctl status firewalld.service
    • 关闭防火墙
    systemctl stop firewalld.service
    • 开机禁用防火墙
    systemctl disable firewalld.service

    物料安装包下载路径为：/opt/app/soft 。

    下载安装物料，地址详见物料清单¶

    # 下载MySQL5.7
wget --no-check-certificate <https://mirrors.bfsu.edu.cn/mysql/downloads/MySQL-5.7/mysql-5.7.35-1.el7.x86_64.rpm-bundle.tar>

    查询并 清理 MySQL或者MariaDB依赖¶

    # 查询MySQL或者MariaDB依赖
rpm -qa | grep -iE 'MySQL|MariaDB'
# 清理MySQL或者MariaDB依赖
rpm -qa | grep -iE 'MySQL|MariaDB'|xargs rpm -e --nodeps

    解压安装¶

    # 解压安装
mkdir mysql && tar -xvf mysql-5.7.35-1.el7.x86_64.rpm-bundle.tar -C mysql && cd mysql
rpm -ivh mysql-community-common-5.7.35-1.el7.x86_64.rpm
rpm -ivh mysql-community-libs-5.7.35-1.el7.x86_64.rpm
rpm -ivh mysql-community-client-5.7.35-1.el7.x86_64.rpm
rpm -ivh mysql-community-server-5.7.35-1.el7.x86_64.rpm --force --nodeps

    启动并检查MySQL服务是否正常¶

    # 启动MySQL服务
service mysqld start

# 检查MySQL服务是否正常
service mysqld status

    修改默认密码¶

    MySQL启动会有一个默认密码在日志中生成，查看默认密码并修改为1qazZSE$

    # 查看默认密码
grep 'temporary password' /var/log/mysqld.log
# 登录mysql
mysql -uroot -p # 修改root密码
ALTER USER 'root'@'localhost' IDENTIFIED BY '1qazZSE$';

    配置允许外部访问¶

    MySQL安装好以后，默认是不允许外部IP进行访问，需要设置允许外部IP访问

    use mysql;
update user set Host='%' where User='root';
flush privileges;
select host,user from mysql.user;

    常见问题¶

    忘记密码

    • 安装模式启动，跳过认证
    # redhat系 >=7 停止mysql服务
systemctl stop mysqld 

# 跳过mysql权限表启动
/usr/bin/mysqld_safe --skip-grant-tables

#无密码登陆
mysql -uroot 
use mysql;
# 设置新密码
update user set authentication_string=password('xxx') where user='root'; 
#刷新权限
flush privileges;
    • 修改配置文件，跳过认证
    # 修改/etc/my.cnf
[mysqld]
skip-grant-tables

# 重启mysql
service mysqld restart

# 无密码登录mysql
mysql -uroot

# 修改密码
update mysql.user set authentication_string=password('xxx') where user='root';

#刷新
flush privileges;

# 删除/etc/my.cnf中的skip-grant-tables配置，重启mysql
service mysqld restart

# 密码登录
mysql -uroot -p
